## 对数器

```cpp

#include <random>

rand() // 0 - RAND_MAX
uniform_real_distrubution() // 0.0-1.0

double rand_double(double l,double r){
    return l + (r - l) * (rand() / RAND_MAX);
}

vector<int> GetRandArr(int maxSize, int maxValue){
    vector<int> arr(rand() % maxSize);
    for (int &x : arr){
        x = rand() % (2 * maxValue - 1) - maxValue; // [-maxValue, maxValue]
    }
    return arr;
}

bool isEqual(vector<int>& a, vector<int>& b){
    if (a.size() != b.size()) return false;
    for (int i = 0; i < a.size(); ++i){
        if (a[i] != b[i]) return false;
    }
    return true;
}
```
