vector<int> GetRandArr(int maxSize, int maxValue){
    vector<int> arr(rand() % maxSize);
    for (int &x : arr){
        x = rand() % (2 * maxValue - 1) - maxValue; // [-maxValue, maxValue]
    }
    return arr;
}

bool isEqual(vector<int>& a, vector<int>& b){
    if (a.size() != b.size()) return false;
    for (int i = 0; i < a.size(); ++i){
        if (a[i] != b[i]) return false;
    }
    return true;
}

## 快读快写

```cpp
// 第一种, 关闭cin和cout 与 stdio的同步，关闭后不能混合使用getchar()和scanf()和cin()和cout()
#define IOS ios::sync_with_stdio(false); cin.tie(nullptr); cout.tie(nullptr)

// 第二种快读快写
inline int read(){
    int x = 0, f = 1;
    char c = getchar();
    while (c < '0' || c > '9'){
        if (c == '-') f = -1;
        c = getchar();
    }
    while (c >= '0' && c <= '9'){
        x = x * 10 + c - '0';
        c = getchar();
    }
    return x * f;
}

inline void write(int x){
    if (x < 0) {putchar('-'); x = -x;}
    if (x >= 10) write(x / 10);
    putchar(x % 10 + '0');
}

inline void writeln(int x){
    write(x);
    putchar('\n');
}
```

## 没思路
不要盯着题目发呆，拿起笔和纸！模拟样例 (Simulate)：像机器人一样，用手一步步模拟样例的输入输出。寻找规律 (Pattern)：在手算的过程中，你是否在重复某种操作？（比如一直在找最大值，或者一直在尝试所有组合）。看数据范围 (Constraints)：如果 $N \le 20$，大概率是 DFS 暴搜 / 状压 DP ($O(2^N)$)。如果 $N \le 1000$，可能是 $O(N^2)$ 的 DP 或简单模拟。如果 $N \le 10^5$，必须想 $O(N \log N)$ (排序/二分) 或 $O(N)$ (贪心/双指针)
